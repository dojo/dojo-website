dojo/router (version 1.8)

A singleton-style instance of dojo/router/RouterBase. See that module for specifics.

Examples

Example 1

router.register("/widgets/:id", function(evt){
    // If "/widgets/3" was matched,
    // evt.params.id === "3"
    xhr.get({
        url: "/some/path/" + evt.params.id,
        load: function(data){
            // ...
        }
    });
});
